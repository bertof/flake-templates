\begin{tikzpicture}[node distance=2em,every node/.style={boxes, fill=sesarlightgreen!50}]
	% Application Layer
	\node (application) [fill=sesarlightorange!50] {Application};

	% Presentation Layer
	\node (presentation) [below of=application] {Presentation};

	% Session Layer
	\node (session) [below of=presentation] {Session};

	% Transport Layer
	\node (transport) [below of=session] {Transport};

	% Network Layer
	\node (network) [below of=transport] {Network};

	% Data Link Layer
	\node (datalink) [below of=network] {Data Link};

	% Physical Layer
	\node (physical) [below of=datalink] {Physical};

	% Connecting lines
	\draw (application) -- (presentation);
	\draw (presentation) -- (session);
	\draw (session) -- (transport);
	\draw (transport) -- (network);
	\draw (network) -- (datalink);
	\draw (datalink) -- (physical);
\end{tikzpicture}
